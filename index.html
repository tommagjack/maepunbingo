    <script>

        // State Management

        let availableNumbers = [];

        let calledNumbers = [];

        let isPlaying = false;

        

        // Settings

        const MAX_NUMBER = 75;

        const SPEEDS = {

            slow: { delay: 3500, rate: 0.65 },

            normal: { delay: 2000, rate: 0.75 },

            fast: { delay: 1000, rate: 0.9 }

        };

        let currentSpeed = SPEEDS.normal;


        // Async controls

        let timeoutId = null;

        let resolveSleep = null;

        let currentUtterance = null;


        // Initialize App

        function init() {

            resetBingo();

        }


        // Convert Number to Thai Words

        function numberToThai(num) {

            const units = ["", "หนึ่ง", "สอง", "สาม", "สี่", "ห้า", "หก", "เจ็ด", "แปด", "เก้า"];

            const tens = ["", "สิบ", "ยี่สิบ", "สามสิบ", "สี่สิบ", "ห้าสิบ", "หกสิบ", "เจ็ดสิบ", "แปดสิบ", "เก้าสิบ"];

            

            if (num < 10) return units[num];

            

            const tenDigit = Math.floor(num / 10);

            const unitDigit = num % 10;

            

            let text = tens[tenDigit];

            

            if (unitDigit === 1) {

                text += "เอ็ด";

            } else if (unitDigit > 1) {

                text += units[unitDigit];

            }

            

            if (num === 10) return "สิบ";

            

            return text;

        }


        // Setup Grid UI

        function renderGrid() {

            const grid = document.getElementById('bingo-grid');

            grid.innerHTML = '';

            for (let i = 1; i <= MAX_NUMBER; i++) {

                const cell = document.createElement('div');

                cell.className = 'cell';

                cell.id = `cell-${i}`;

                cell.innerText = i;

                if (calledNumbers.includes(i)) {

                    cell.classList.add('called');

                }

                grid.appendChild(cell);

            }

        }


        // Update Counter UI

        function updateCounter() {

            document.getElementById('number-counter').innerText = `ออกแล้ว ${calledNumbers.length} / ${MAX_NUMBER}`;

        }


        // Update Buttons UI

        function updateUI() {

            document.getElementById('btn-start').disabled = isPlaying || availableNumbers.length === 0;

            document.getElementById('btn-stop').disabled = !isPlaying;

            document.getElementById('btn-bingo').disabled = calledNumbers.length === 0;

            

            const displayCard = document.getElementById('display-card');

            if (isPlaying) {

                displayCard.classList.add('active');

            } else {

                displayCard.classList.remove('active');

            }

        }


        // Set Speech Speed

        function setSpeed(type) {

            currentSpeed = SPEEDS[type];

            document.querySelectorAll('.btn-speed').forEach(btn => btn.classList.remove('active'));

            document.getElementById(`speed-${type}`).classList.add('active');

        }


        // Promise-based sleep function (cancellable)

        function sleep(ms) {

            return new Promise(resolve => {

                resolveSleep = resolve;

                timeoutId = setTimeout(() => {

                    resolveSleep = null;

                    resolve();

                }, ms);

            });

        }


        function cancelSleep() {

            if (timeoutId) {

                clearTimeout(timeoutId);

                timeoutId = null;

            }

            if (resolveSleep) {

                resolveSleep();

                resolveSleep = null;

            }

        }


        // Promise-based Speech function

        function speak(text, rate) {

            return new Promise(resolve => {

                if (!('speechSynthesis' in window)) {

                    return resolve();

                }


                window.speechSynthesis.cancel(); // Cancel ongoing speech

                

                currentUtterance = new SpeechSynthesisUtterance(text);

                currentUtterance.lang = 'th-TH';

                currentUtterance.rate = rate;

                

                currentUtterance.onend = () => {

                    currentUtterance = null;

                    resolve();

                };

                

                currentUtterance.onerror = () => {

                    currentUtterance = null;

                    resolve();

                };

                

                window.speechSynthesis.speak(currentUtterance);

            });

        }


        // Main Loop

        async function startAuto() {

            if (isPlaying || availableNumbers.length === 0) return;

            isPlaying = true;

            updateUI();


            // Wake up speech engine (browser requirement)

            if ('speechSynthesis' in window) {

                window.speechSynthesis.resume();

            }


            while (isPlaying && availableNumbers.length > 0) {

                // Randomly pick a number

                const randIndex = Math.floor(Math.random() * availableNumbers.length);

                const num = availableNumbers.splice(randIndex, 1)[0];

                calledNumbers.push(num);


                // Update UI state inside loop so buttons like Bingo become active

                updateCounter();

                updateUI();


                // Update Display

                document.getElementById('current-number').innerText = num;

                const cell = document.getElementById(`cell-${num}`);

                if (cell) cell.classList.add('called');


                // Prepare Thai Speech

                const thaiWord = numberToThai(num);

                const textToSpeak = `เลข ${thaiWord} ... ย้ำอีกครั้ง เลข ${thaiWord}`; 


                // Speak and await completion

                await speak(textToSpeak, currentSpeed.rate);


                // Check if stopped during speech

                if (!isPlaying) break;


                // Wait based on speed setting

                await sleep(currentSpeed.delay);

            }


            // End of game

            if (availableNumbers.length === 0) {

                isPlaying = false;

                document.getElementById('current-number').innerText = "จบเกม";

                await speak("จบเกมบิงโกแล้วครับ", currentSpeed.rate);

            }

            updateUI();

        }


        // Stop functionality

        function stopBingo() {

            isPlaying = false;

            if ('speechSynthesis' in window) {

                window.speechSynthesis.cancel();

            }

            cancelSleep();

            updateUI();

        }


        // Reset functionality

        function resetBingo() {

            stopBingo();

            availableNumbers = Array.from({length: MAX_NUMBER}, (_, i) => i + 1);

            calledNumbers = [];

            document.getElementById('current-number').innerText = "-";

            updateCounter();

            renderGrid();

            updateUI();

        }


        // Announce Bingo functionality

        async function announceBingo() {

            stopBingo();

            document.getElementById('current-number').innerText = "บิงโก!";

            document.getElementById('display-card').classList.add('active');

            await speak("บิงโก! บิงโก! มีผู้ชนะแล้วครับ", currentSpeed.rate);

        }


        // Fullscreen Toggle

        function toggleFullscreen() {

            if (!document.fullscreenElement) {

                document.documentElement.requestFullscreen().catch(err => {

                    console.log(`Error attempting to enable fullscreen: ${err.message}`);

                });

            } else {

                if (document.exitFullscreen) {

                    document.exitFullscreen();

                }

            }

        }


        // Start initialization on window load

        window.onload = init;

    </script>
